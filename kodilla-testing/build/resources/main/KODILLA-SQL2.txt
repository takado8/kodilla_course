SELECT IL.NAME, I.*
FROM issueslists IL
JOIN issues I on IL.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM issues I
JOIN users U ON I.USER_ID_ASSIGNED_TO = U.ID;


SELECT COUNT(*), U.FIRSTNAME, U.LASTNAME
FROM issues I
JOIN users U ON I.USER_ID_ASSIGNED_TO = U.ID
GROUP BY U.ID
HAVING COUNT(*) > 1;
